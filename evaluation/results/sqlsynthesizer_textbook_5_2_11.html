Our Solution


SELECT
    max(T0.sname),
    max(T2.cost) 
FROM
    input3 AS T0 
LEFT JOIN
    (
        SELECT
            part_id 
        FROM
            input2 
        WHERE
            color = 'green' 
            OR color = 'red'
    ) AS T1 
        ON T0.supplier_key = T1.part_id 
JOIN
    (
        SELECT
            cost,
            string_agg(supplier_key, ', ') AS concat_comma_supplier_key 
        FROM
            input1 
        GROUP BY
            cost
    ) AS T2 
        ON T0.supplier_key = T2.concat_comma_supplier_key 
GROUP BY
    T0.supplier_key 
ORDER BY
    max(T0.sname) ASC
